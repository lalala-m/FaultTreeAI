### 2.4 FaultTreeViewer 边方向修正

**文件：** `frontend/src/components/FaultTreeViewer.jsx`

当前边方向是反的。故障树中，底事件在底部，逻辑门向上汇聚到顶事件。
当前代码把 `g.children` 当作 source，`g.output_node` 当作 target —— 方向全反了。

修正 edges 构建逻辑：

```javascript
const edges = useMemo(() => {
  const gates = tree.gates_json || []
  return gates.flatMap((g) =>
    (g.input_nodes || []).map((childId, i) => ({
      id: `${g.id}_${i}`,
      source: g.output_node,   // ✅ 逻辑门是源（上游）
      target: childId,        // ✅ 子节点是目标（下游）
      animated: g.type === 'OR',
      label: g.type,
      style: { stroke: NODE_COLORS[g.type?.toLowerCase()] || '#999' },
      markerEnd: {
        type: MarkerType.ArrowClosed,
        color: NODE_COLORS[g.type?.toLowerCase()] || '#999',
      },
    }))
  )
}, [tree])
```

同时修正节点数据读取：

const nodes = useMemo(() => {

**新增文件：** `frontend/src/components/TreeEditor.jsx`

核心实现：
```javascript
import React, { useCallback, useState } from 'react'
import ReactFlow, {
  addNode, removeNodes, updateEdge,
  useNodesState, useEdgesState,
  Controls, Background,
} from 'reactflow'
import 'reactflow/dist/style.css'
import { ContextMenu, MenuItem } from './ContextMenu.jsx'  // 右键菜单

export default function TreeEditor({ initialNodes, initialEdges, onSave }) {
  const [nodes, setNodes, onNodesChange] = useNodesState(initialNodes)
  const [edges, setEdges, onEdgesChange] = useEdgesState(initialEdges)

  // 添加节点
  const addNodeToGraph = useCallback((parentId, nodeType) => {
    const newNode = {
      id: `N${Date.now()}`,
      type: nodeType,           // 'basic' | 'gate' | 'intermediate'
      position: { x: 0, y: 100 },
      data: { label: '新节点', type: nodeType },
    }
    setNodes(nds => addNode(nds, newNode, parentId))
  }, [setNodes])

  // 删除节点
  const deleteNode = useCallback((nodeId) => {
    setNodes(nds => nds.filter(n => n.id !== nodeId))
    setEdges(eds => eds.filter(e => e.source !== nodeId && e.target !== nodeId))
  }, [setNodes, setEdges])

  // 右键菜单
  const onNodeContextMenu = useCallback((event, node) => {
    event.preventDefault()
    // 显示菜单：添加子节点 / 修改门类型 / 删除节点
  }, [])

  // 保存
  const handleSave = async () => {
    // 将 nodes/edges 转换回 { nodes, gates } 结构，调用后端保存
    await onSave({ nodes, edges })
  }

  return (
    <>
      <ReactFlow
        nodes={nodes}
        edges={edges}
        onNodesChange={onNodesChange}
        onEdgesChange={onEdgesChange}
        onNodeContextMenu={onNodeContextMenu}
        fitView
      >
        <Controls />
        <Background />
      </ReactFlow>
      <button onClick={handleSave}>保存修改</button>
    </>
  )
}
```

from fastapi import APIRouter, HTTPException, Depends
from sqlalchemy.ext.asyncio import AsyncSession
from backend.core.database.connection import get_db
from backend.core.database.models import FaultTree
from backend.core.validator.checker import validate_fault_tree
from backend.models.schemas import FaultTree as FT, GenerateResponse
from uuid import UUID

router = APIRouter(prefix="/api/edit", tags=["故障树编辑"])

@router.put("/{tree_id}", response_model=GenerateResponse)
async def update_tree(tree_id: str, data: dict, db: AsyncSession = Depends(get_db)):
    """保存编辑后的故障树（前端传入 nodes + gates，存入数据库）"""
    result = await db.execute(
        select(DBFaultTree).where(DBFaultTree.tree_id == UUID(tree_id))
    )
    tree = result.scalar_one_or_none()
    if not tree:
        raise HTTPException(status_code=404, detail="故障树不存在")

    tree.nodes_json = data['nodes']
    tree.gates_json = data['gates']
    await db.commit()

    return GenerateResponse(
        fault_tree=FT(**data['ft']),
        mcs=data.get('mcs', []),
        importance=data.get('importance', []),
        validation_issues=data.get('issues', []),
    )

@router.post("/pdf")
def export_pdf(fault_tree: FaultTree, mcs: list = None):
    from reportlab.lib.pagesizes import A4
    from reportlab.lib import colors
    from reportlab.lib.units import cm
    from reportlab.platypus import SimpleDocTemplate, Table, TableStyle, Paragraph, Spacer

    buffer = io.BytesIO()
    doc = SimpleDocTemplate(buffer, pagesize=A4)
    story = []

    story.append(Paragraph("故障树分析报告", ParagraphStyle('Title')))
    story.append(Spacer(1, 0.5 * cm))
    story.append(Paragraph(f"顶事件：{fault_tree.top_event}", ParagraphStyle('H2')))
    story.append(Paragraph(f"分析摘要：{fault_tree.analysis_summary}"))

    if mcs:
        story.append(Spacer(1, 0.3 * cm))
        story.append(Paragraph("最小割集：", ParagraphStyle('H2')))
        for i, cut in enumerate(mcs, 1):
            story.append(Paragraph(f"  MCS {i}: {' + '.join(cut)}", ParagraphStyle('Body')))

    doc.build(story)
    buffer.seek(0)
    return StreamingResponse(
        buffer,
        media_type="application/pdf",
        filename="故障树分析报告.pdf",
    )

### 6.3 Redis 缓存

```python
import redis.asyncio as redis

redis_client = redis.from_url("redis://localhost")

async def cached_retrieve(query, doc_ids=None):
    cache_key = f"rag:{hash((query, tuple(doc_ids or [])))}"
    cached = await redis_client.get(cache_key)
    if cached:
        return json.loads(cached)
    results = await retrieve(query, doc_ids=doc_ids)
    await redis_client.setex(cache_key, 3600, json.dumps(results))  # 1小时缓存
    return results
```
